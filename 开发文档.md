```python
from sqlalchemy.orm import sessionmaker
from app.config import DATABASE_URL

# 创建引擎（Python和MySQL之间的桥梁）
engine = create_engine(DATABASE_URL, pool_pre_ping=True)

# 创建会话工厂（每次数据库操作从这里拿连接）
SessionLocal = sessionmaker(autocommit=False, autoflush=False, bind=engine)

# 基类（所有数据库表都继承这个）
Base = declarative_base()

def get_db():
    """获取数据库连接，用完后自动关闭"""
    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()

def init_db():
    """启动时自动创建所有表"""
    from app.models import prediction, match
    Base.metadata.create_all(bind=engine)
```

### 4.3 创建历史API

创建 `D:\足球预测系统\backend\app\api\history.py`：

```python
from fastapi import APIRouter, Depends, Query
from sqlalchemy.orm import Session
from app.database import get_db
from app.models.prediction import PredictionHistory
from datetime import datetime, timedelta

router = APIRouter()

@router.get("/list")
async def get_history_list(
    start_date: str = Query(None),
    end_date: str = Query(None),
    league: str = Query(None),
    team: str = Query(None),
    page: int = Query(1),
    page_size: int = Query(20),
    db: Session = Depends(get_db)
):
    """历史预测列表"""
    query = db.query(PredictionHistory)

    # 日期筛选
    if start_date:
        query = query.filter(PredictionHistory.match_date >= start_date)
    else:
        query = query.filter(
            PredictionHistory.match_date >= datetime.now() - timedelta(days=30))

    if end_date:
        query = query.filter(PredictionHistory.match_date <= end_date)

    # 联赛筛选
    if league:
        query = query.filter(PredictionHistory.league == league)

    # 队伍筛选
    if team:
        query = query.filter(
            (PredictionHistory.home_team.contains(team)) |
            (PredictionHistory.away_team.contains(team))
        )

    total = query.count()
    records = query.order_by(PredictionHistory.match_date.desc())\
        .offset((page - 1) * page_size).limit(page_size).all()

    return {
        "total": total,
        "page": page,
        "page_size": page_size,
        "records": [r.to_dict() for r in records]
    }

@router.get("/statistics")
async def get_statistics(db: Session = Depends(get_db)):
    """统计数据"""
    records = db.query(PredictionHistory).filter(
        PredictionHistory.is_correct.in_([1, 2])
    ).all()

    total = len(records)
    correct = sum(1 for r in records if r.is_correct == 1)
    wrong = sum(1 for r in records if r.is_correct == 2)

    # 按预测类型统计
    type_stats = {"home": [0, 0], "draw": [0, 0], "away": [0, 0]}
    for r in records:
        if r.prediction_result in type_stats:
            type_stats[r.prediction_result][0] += 1
            if r.is_correct == 1:
                type_stats[r.prediction_result][1] += 1

    return {
        "summary": {
            "total": total,
            "correct": correct,
            "wrong": wrong,
            "accuracy": round(correct/(correct+wrong)*100, 1) if (correct+wrong) else 0,
        },
        "prediction_type_accuracy": [
            {"type": t, "name": {"home": "主胜", "draw": "平局", "away": "客胜"}[t],
             "total": d[0],
             "accuracy": round(d[1]/d[0]*100, 1) if d[0] else 0}
            for t, d in type_stats.items()
        ]
    }
```

### 5.1 预测引擎架构

```
预测引擎 = 10个维度 × 各自权重 = 最终概率

10个维度：
  1. 球队实力 (20%)
  2. 近期状态 (15%)
  3. 历史交锋 (10%)
  4. 赔率分析 (8%)
  5. 盘口趋势 (8%)
  6. 外部预测 (动态20-35%)
  7. 新闻因素 (8%)
  8. 主客场 (8%)
  9. 进球效率 (8%)
  10. 赛程体能 (7%)
```

### 5.3 实力维度计算

```python
def _strength_probability(self, diff):
    """根据实力差距计算胜平负概率"""
    # diff = 主队实力 - 客队实力

    if abs(diff) < 3:
        # 实力非常接近 → 平局概率最高
        return {'home': 0.34, 'draw': 0.38, 'away': 0.28}
    elif abs(diff) < 6:
        # 有一定差距 → 强队略占优
        return {'home': 0.38, 'draw': 0.34, 'away': 0.28}
    elif abs(diff) < 12:
        # 差距中等
        r = (abs(diff) - 6) / 6
        return {'home': 0.42 + r * 0.16, 'draw': 0.32 - r * 0.06,
                'away': 0.26 - r * 0.10}
    else:
        # 差距很大 → 强队稳赢
        return {'home': 0.58, 'draw': 0.26, 'away': 0.16}
```

### 5.4 多维度加权求和

```python
# 10个维度各自计算概率后，按权重相加
home_p = (strength['home'] * 0.20 + form['home'] * 0.15 +
          h2h['home'] * 0.10 + odds['home'] * 0.08 +
          market['home'] * 0.08 + external['home'] * ext_weight +
          news['home'] * 0.08 + home_away['home'] * 0.08 +
          goals['home'] * 0.08 + fatigue['home'] * 0.07)

# 同理计算 draw_p 和 away_p

# 归一化（确保三个数加起来=100%）
total = home_p + draw_p + away_p
final = {'home': home_p/total*100, 'draw': draw_p/total*100, 'away': away_p/total*100}

# 选最大概率
best = max(final, key=final.get)
```

```python
from selenium import webdriver
from selenium.webdriver.chrome.options import Options
from webdriver_manager.chrome import ChromeDriverManager

def fetch_page(url):
    """用Chrome浏览器打开网页，获取渲染后的HTML"""
    options = Options()
    options.add_argument('--headless=new')  # 隐藏浏览器窗口
    options.add_argument('--disable-gpu')

    driver = webdriver.Chrome(
        service=Service(ChromeDriverManager().install()),
        options=options
    )

    # 隐藏自动化特征（反反爬）
    driver.execute_cdp_cmd('Page.addScriptToEvaluateOnNewDocument', {
        'source': '''
            Object.defineProperty(navigator, "webdriver", {
                get: () => undefined
            })
        '''
    })

    driver.get(url)
    time.sleep(5)  # 等JS加载
    html = driver.page_source
    driver.quit()
    return html
```

### 7.2 体彩官网抓取

```python
def scrape_sporttery():
    html = fetch_page('https://www.sporttery.cn/jc/zqsgkj/')

    # 用正则提取比分数据
    import re
    results = []
    rows = re.findall(r'<tr[^>]*>(.*?)</tr>', html, re.DOTALL)

    for row in rows:
        # 提取每个单元格
        cells = re.findall(r'<td[^>]*>(.*?)</td>', row, re.DOTALL)
        texts = [re.sub(r'<[^>]+>', '', c).strip() for c in cells]

        # 找日期、队名、比分
        date = match_num = home = away = score = ''
        for t in texts:
            if re.match(r'\d{4}-\d{2}-\d{2}', t):
                date = t
            elif 'VS' in t:
                parts = t.split('VS')
                home = parts[0].replace('(-1)', '').strip()
                away = parts[1].replace('(+1)', '').strip()
            elif re.match(r'\d+:\d+', t):
                score = t

        if date and home and score:
            results.append({'date': date, 'home': home,
                           'away': away, 'score': score})
    return results
```

## 十、常见问题

**Q: 为什么有些比赛预测不对？**
A: 足球有随机性。专业机构准确率也只有50-60%。51.1%已优于随机猜测。

**Q: 如何提高准确率？**
A: (1)每天打开累积真实数据 (2)添加更多球队评分 (3)接入更多预测源

**Q: 为什么外部数据匹配率不高？**
A: 竞彩比赛以日职/芬超为主，外部源主要覆盖五大联赛。8月新赛季会改善。

**Q: 系统吃多少内存？**
A: 约200MB（Python进程+Selenium Chrome），普通电脑完全够用。

**Q: 数据库密码忘了怎么办？**
A: 修改 config.py 中的 DATABASE_URL，把 123456 改成你的密码。
